<script>
        function togglePassword() {
            const passInput = document.getElementById('password');
            const passIcon = document.getElementById('passIcon');
            
            if (passInput.type === 'password') {
                passInput.type = 'text';
                passIcon.innerText = 'visibility_off';
            } else {
                passInput.type = 'password';
                passIcon.innerText = 'visibility';
            }
        }

        // Card micro-interaction
        const card = document.querySelector('.glass-panel');
        document.addEventListener('mousemove', (e) => {
            if (window.innerWidth < 768) return;
            const xAxis = (window.innerWidth / 2 - e.pageX) / 80;
            const yAxis = (window.innerHeight / 2 - e.pageY) / 80;
            card.style.transform = `perspective(1000px) rotateY(${xAxis}deg) rotateX(${yAxis}deg)`;
        });

        document.addEventListener('mouseleave', () => {
            card.style.transform = `perspective(1000px) rotateY(0deg) rotateX(0deg)`;
        });
    </script>
